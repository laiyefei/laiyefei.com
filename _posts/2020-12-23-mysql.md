# 安装过程

~~~
1. sudo rpm -Uvh https://dev.mysql.com/get/mysql80-community-release-el7-3.noarch.rpm
2. sudo yum --enablerepo=mysql80-community install mysql-community-server
3. systemctl start mysqld
4. systemctl status mysqld
5. systemctl stop mysqld
6. systemctl restart mysqld
7. netstat -nlpt
8. 
~~~
